rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function to check if user is an admin
    function isAdmin() {
      return isAuthenticated() && exists(/databases/$(database)/documents/admins/$(request.auth.uid));
    }
    
    match /admins/{adminId} {
      allow read: if isAuthenticated();
      allow create: if true; // Allow creation during registration
      allow update: if isAuthenticated() && request.auth.uid == adminId;
      allow delete: if false; // Prevent deletion
    }
    
    match /users/{userId} {
      allow create: if true; // Allow creation during registration
      allow read: if isAuthenticated(); // Any authenticated user can read
      allow update: if isAuthenticated() && (isAdmin() || request.auth.uid == userId);
      allow delete: if isAdmin(); // Only admins can delete users
    }
    
    match /drivers/{driverId} {
      allow read: if isAuthenticated(); // Any authenticated user can read drivers
      allow create: if true; // Allow creation during registration
      allow update: if isAuthenticated() && (isAdmin() || request.auth.uid == driverId);
      allow delete: if isAdmin(); // Only admins can delete drivers
    }
    
    // Orders collection
    match /orders/{orderId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if isAuthenticated();
      allow delete: if isAuthenticated();
    }
    
    // Driver applications collection
    match /driver_applications/{applicationId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if isAuthenticated();
      allow delete: if isAuthenticated();
    }
    
    match /settings/{settingId} {
      allow read: if isAuthenticated(); // All users need to read settings
      allow write: if isAuthenticated(); // Allow writes for admin operations
    }
    
    // System logs collection
    match /system_logs/{logId} {
      allow create: if isAuthenticated();
      allow read: if isAuthenticated();
      allow delete: if isAuthenticated();
      allow update: if false;
    }
    
    // Notifications collection
    match /notifications/{notificationId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if isAuthenticated();
      allow delete: if isAuthenticated();
    }
    
    // Complaints collection for fraud detection
    match /complaints/{complaintId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Chats collection - for user-driver conversations
    match /chats/{chatId} {
      // Allow authenticated users to read chats they're part of, or admins to read all
      allow read: if isAuthenticated();
      // Allow authenticated users to create chats
      allow create: if isAuthenticated();
      // Allow participants or admins to update chat
      allow update: if isAuthenticated();
      // Only admins can delete chats
      allow delete: if isAdmin();
      
      // Messages subcollection within chats
      match /messages/{messageId} {
        allow read: if isAuthenticated();
        allow create: if isAuthenticated();
        allow update: if isAuthenticated();
        allow delete: if isAdmin();
      }
    }
    
    // Default deny all other collections
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
