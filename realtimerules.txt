{
  "rules": {
    // Users can read and write their own data
    "users": {
      "$uid": {
        ".read": "$uid === auth.uid",
        ".write": "$uid === auth.uid"
      }
    },
    
    // Orders - users can read/write their own orders
    "orders": {
      "$orderId": {
        ".read": "data.child('userId').val() === auth.uid || data.child('driverId').val() === auth.uid",
        ".write": "data.child('userId').val() === auth.uid || data.child('driverId').val() === auth.uid",
        ".validate": "newData.hasChildren(['userId', 'serviceType', 'status', 'createdAt'])"
      }
    },
    
    // Chats - only participants can access
    "chats": {
      "$chatId": {
        ".read": "data.child('participants').child(auth.uid).exists()",
        ".write": "data.child('participants').child(auth.uid).exists()"
      }
    },
    
    // Payments - users can read/write their own payments
    "payments": {
      "$paymentId": {
        ".read": "data.child('userId').val() === auth.uid",
        ".write": "data.child('userId').val() === auth.uid"
      }
    },
    
    // Driver locations - public read for tracking, drivers can write their own
    "driverLocations": {
      "$driverId": {
        ".read": true,
        ".write": "$driverId === auth.uid"
      }
    },
    
    // Services - public read, admin write only
    "services": {
      ".read": true,
      ".write": false
    },
    
    // Default deny all other paths
    "$other": {
      ".read": false,
      ".write": false
    }
  }
}
